CREATE DATABASE IF NOT EXISTS hospitalmanagement;
USE hospitalmanagement;

CREATE TABLE admin_appointment_history (
    id INT(11) NOT NULL AUTO_INCREMENT,
    doctor_report_id INT(11) DEFAULT NULL,
    patient_name VARCHAR(100) NOT NULL,
    patient_id VARCHAR(20) NOT NULL,
    report_data TEXT,
    doctor_notes TEXT,
    status VARCHAR(50) DEFAULT 'Received from Doctor',
    sent_by VARCHAR(100),
    sent_to VARCHAR(100),
    sent_date DATETIME,
    approved_by VARCHAR(100),
    approved_role VARCHAR(50),
    approved_date DATETIME,
    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE appointments (
    appointment_id VARCHAR(15) NOT NULL,
    patient_name VARCHAR(100) NOT NULL,
    doctor_name VARCHAR(100) NOT NULL,
    treatment VARCHAR(100) NOT NULL,
    status VARCHAR(20) NOT NULL,
    appointment_date VARCHAR(30),
    PRIMARY KEY (appointment_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE nurse_appointment_history (
    id INT(11) NOT NULL AUTO_INCREMENT,
    patient_name VARCHAR(100) NOT NULL,
    patient_id VARCHAR(20) NOT NULL,
    report_data TEXT,
    status VARCHAR(50) DEFAULT 'Draft',
    sent_by VARCHAR(100),
    sent_to VARCHAR(100),
    sent_date DATETIME,
    approved_by VARCHAR(100),
    approved_role VARCHAR(50),
    approved_date DATETIME,
    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE doctor_appointment_history (
    id INT(11) NOT NULL AUTO_INCREMENT,
    nurse_report_id INT(11),
    patient_name VARCHAR(100) NOT NULL,
    patient_id VARCHAR(20) NOT NULL,
    report_data TEXT,
    status VARCHAR(50) DEFAULT 'Received from Nurse',
    doctor_notes TEXT,
    sent_by VARCHAR(100),
    sent_to VARCHAR(100),
    sent_date DATETIME,
    approved_by VARCHAR(100),
    approved_role VARCHAR(50),
    approved_date DATETIME,
    PRIMARY KEY (id),
    CONSTRAINT fk_doctor_nurse
        FOREIGN KEY (nurse_report_id)
        REFERENCES nurse_appointment_history(id)
        ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE employees (
    id VARCHAR(10) NOT NULL,
    name VARCHAR(50) NOT NULL,
    username VARCHAR(150) NOT NULL,
    password VARCHAR(50) NOT NULL,
    role VARCHAR(20) NOT NULL,
    status VARCHAR(20) NOT NULL,
    age INT(11) NOT NULL,
    department VARCHAR(50),
    email VARCHAR(100),
    phone VARCHAR(20),
    profile_image VARCHAR(255)
        DEFAULT 'src/profile_images/default.png',
    PRIMARY KEY (id),
    UNIQUE KEY uk_username (username)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE inventory (
    id INT(11) NOT NULL AUTO_INCREMENT,
    category VARCHAR(50) NOT NULL,
    item VARCHAR(100) NOT NULL,
    quantity INT(11) NOT NULL,
    price DOUBLE NOT NULL,
    status VARCHAR(30) NOT NULL,
    expiry DATE NOT NULL,
    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE logistics (
    order_id VARCHAR(15) NOT NULL,
    item VARCHAR(100) NOT NULL,
    amount DOUBLE NOT NULL,
    status VARCHAR(20) NOT NULL,
    is_checked_in TINYINT(1) DEFAULT 0,
    PRIMARY KEY (order_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE medical_requests (
    request_id INT(11) NOT NULL AUTO_INCREMENT,
    category VARCHAR(50) NOT NULL,
    item_name VARCHAR(100) NOT NULL,
    quantity INT(11) NOT NULL,
    billing_amount DECIMAL(10,2) DEFAULT 0.00,
    status VARCHAR(30) DEFAULT 'Pending',
    requested_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    requested_by VARCHAR(20) NOT NULL,
    PRIMARY KEY (request_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE patients (
    id VARCHAR(15) NOT NULL,
    name VARCHAR(100) NOT NULL,
    room VARCHAR(20),
    doctor VARCHAR(100),
    nurse VARCHAR(100),
    status VARCHAR(30) NOT NULL,
    admission_date VARCHAR(30),

    address VARCHAR(255) DEFAULT '',
    contact_number VARCHAR(20) DEFAULT '',

    allergies TEXT,
    conditions TEXT,
    medications TEXT,

    emergency_name VARCHAR(100) DEFAULT '',
    emergency_rel VARCHAR(100) DEFAULT '',
    emergency_num VARCHAR(20) DEFAULT '',

    blood_type VARCHAR(10) DEFAULT 'Pending',
    age INT(11) DEFAULT 0,

    notes TEXT,

    surgical_history TEXT,
    family_history TEXT,
    lifestyle_diet TEXT,
    medical_management TEXT,

    diet_breakfast TEXT,
    diet_lunch TEXT,
    diet_dinner TEXT,

    bp VARCHAR(10),
    hr VARCHAR(10),
    temp VARCHAR(10),
    spo2 VARCHAR(10),

    general_appearance TEXT,
    cardiovascular TEXT,
    respiratory TEXT,
    neurological TEXT,

    PRIMARY KEY (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;


CREATE TABLE prescriptions (
    prescription_id INT(11) NOT NULL AUTO_INCREMENT,
    patient_id VARCHAR(50) NOT NULL,
    patient_name VARCHAR(150) NOT NULL,
    drug_name VARCHAR(150) NOT NULL,
    dosage VARCHAR(50) NOT NULL,
    frequency INT(11) NOT NULL,
    duration INT(11) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) DEFAULT 'Pending',
    PRIMARY KEY (prescription_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
