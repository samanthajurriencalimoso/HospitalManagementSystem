CREATE DATABASE `hospitalmanagement`;

CREATE TABLE `admin_appointment_history` (
  `id` int(11) NOT NULL,
  `doctor_report_id` int(11) DEFAULT NULL,
  `patient_name` varchar(100) NOT NULL,
  `patient_id` varchar(20) NOT NULL,
  `report_data` text DEFAULT NULL,
  `doctor_notes` text DEFAULT NULL,
  `status` varchar(50) DEFAULT 'Received from Doctor',
  `sent_by` varchar(100) DEFAULT NULL,
  `sent_to` varchar(100) DEFAULT NULL,
  `sent_date` datetime DEFAULT NULL,
  `approved_by` varchar(100) DEFAULT NULL,
  `approved_role` varchar(50) DEFAULT NULL,
  `approved_date` datetime DEFAULT NULL
) 

CREATE TABLE `appointments` (
  `appointment_id` varchar(15) NOT NULL,
  `patient_name` varchar(100) NOT NULL,
  `doctor_name` varchar(100) NOT NULL,
  `treatment` varchar(100) NOT NULL,
  `status` varchar(20) NOT NULL,
  `appointment_date` varchar(30) DEFAULT NULL
) 

CREATE TABLE `doctor_appointment_history` (
  `id` int(11) NOT NULL,
  `nurse_report_id` int(11) DEFAULT NULL,
  `patient_name` varchar(100) NOT NULL,
  `patient_id` varchar(20) NOT NULL,
  `report_data` text DEFAULT NULL,
  `status` varchar(50) DEFAULT 'Received from Nurse',
  `doctor_notes` text DEFAULT NULL,
  `sent_by` varchar(100) DEFAULT NULL,
  `sent_to` varchar(100) DEFAULT NULL,
  `sent_date` datetime DEFAULT NULL,
  `approved_by` varchar(100) DEFAULT NULL,
  `approved_role` varchar(50) DEFAULT NULL,
  `approved_date` datetime DEFAULT NULL
) 


CREATE TABLE `employees` (
  `id` varchar(10) NOT NULL,
  `name` varchar(50) NOT NULL,
  `username` varchar(150) NOT NULL,
  `password` varchar(50) NOT NULL,
  `role` varchar(20) NOT NULL,
  `status` varchar(20) NOT NULL,
  `age` int(11) NOT NULL,
  `department` varchar(50) DEFAULT NULL,
  `email` varchar(100) DEFAULT NULL,
  `phone` varchar(20) DEFAULT NULL,
  `profile_image` varchar(255) DEFAULT 'src/profile_images/default.png'
)

CREATE TABLE `inventory` (
  `id` int(11) NOT NULL,
  `category` varchar(50) NOT NULL,
  `item` varchar(100) NOT NULL,
  `quantity` int(11) NOT NULL,
  `price` double NOT NULL,
  `status` varchar(30) NOT NULL,
  `expiry` date NOT NULL
) 

CREATE TABLE `logistics` (
  `order_id` varchar(15) NOT NULL,
  `item` varchar(100) NOT NULL,
  `amount` double NOT NULL,
  `status` varchar(20) NOT NULL,
  `is_checked_in` tinyint(1) DEFAULT 0
) 

CREATE TABLE `medical_requests` (
  `request_id` int(11) NOT NULL,
  `category` varchar(50) NOT NULL,
  `item_name` varchar(100) NOT NULL,
  `quantity` int(11) NOT NULL,
  `billing_amount` decimal(10,2) DEFAULT 0.00,
  `status` varchar(30) DEFAULT 'Pending',
  `requested_at` timestamp NOT NULL DEFAULT current_timestamp(),
  `requested_by` varchar(20) NOT NULL DEFAULT ''
) 

CREATE TABLE `nurse_appointment_history` (
  `id` int(11) NOT NULL,
  `patient_name` varchar(100) NOT NULL,
  `patient_id` varchar(20) NOT NULL,
  `report_data` text DEFAULT NULL,
  `status` varchar(50) DEFAULT 'Draft',
  `sent_by` varchar(100) DEFAULT NULL,
  `sent_to` varchar(100) DEFAULT NULL,
  `sent_date` datetime DEFAULT NULL,
  `approved_by` varchar(100) DEFAULT NULL,
  `approved_role` varchar(50) DEFAULT NULL,
  `approved_date` datetime DEFAULT NULL
) 

CREATE TABLE `patients` (
  `id` varchar(15) NOT NULL,
  `name` varchar(100) NOT NULL,
  `room` varchar(20) DEFAULT NULL,
  `doctor` varchar(100) DEFAULT NULL,
  `nurse` varchar(100) DEFAULT NULL,
  `status` varchar(30) NOT NULL,
  `admission_date` varchar(30) DEFAULT NULL,
  `address` varchar(255) DEFAULT '',
  `contact_number` varchar(20) DEFAULT '',
  `allergies` text DEFAULT '',
  `conditions` text DEFAULT '',
  `medications` text DEFAULT '',
  `emergency_name` varchar(100) DEFAULT '',
  `emergency_rel` varchar(100) DEFAULT '',
  `emergency_num` varchar(20) DEFAULT '',
  `blood_type` varchar(10) DEFAULT 'Pending',
  `age` int(11) DEFAULT 0,
  `notes` text DEFAULT '',
  `surgical_history` text DEFAULT NULL,
  `family_history` text DEFAULT NULL,
  `lifestyle_diet` text DEFAULT NULL,
  `medical_management` text DEFAULT NULL,
  `diet_breakfast` text DEFAULT NULL,
  `diet_lunch` text DEFAULT NULL,
  `diet_dinner` text DEFAULT NULL,
  `bp` varchar(10) DEFAULT NULL,
  `hr` varchar(10) DEFAULT NULL,
  `temp` varchar(10) DEFAULT NULL,
  `spo2` varchar(10) DEFAULT NULL,
  `general_appearance` text DEFAULT NULL,
  `cardiovascular` text DEFAULT NULL,
  `respiratory` text DEFAULT NULL,
  `neurological` text DEFAULT NULL
) 

CREATE TABLE `prescriptions` (
  `prescription_id` int(11) NOT NULL,
  `patient_id` varchar(50) NOT NULL,
  `patient_name` varchar(150) NOT NULL,
  `drug_name` varchar(150) NOT NULL,
  `dosage` varchar(50) NOT NULL,
  `frequency` int(11) NOT NULL,
  `duration` int(11) NOT NULL,
  `created_at` timestamp NOT NULL DEFAULT current_timestamp(),
  `status` varchar(20) DEFAULT 'Pending'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
